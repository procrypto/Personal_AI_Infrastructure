    34→type LocalApiOverrides = Record<string, boolean>;
    35→
    36→// Optional, gitignored local overrides file:
    37→// Place a file at apps/web-terminal/.local/local-api.overrides.ts exporting:
    38→//   export default { trades: true, wallet_tracker: true, search: true };
    39→// It is only loaded in local env; missing files are ignored.
    40→const localOverrideModules: Record<string, { default?: LocalApiOverrides }> =
    41→  IS_LOCAL_ENV
    42→    ? import.meta.glob("../.local/local-api.overrides.{ts,js}", { eager: true })
    43→    : {};
    44→
    45→const LOCAL_API_FILE_OVERRIDES: LocalApiOverrides = Object.values(
    46→  localOverrideModules
    47→).reduce((acc, mod) => {
    48→  if (mod?.default && typeof mod.default === "object") {
    49→    Object.entries(mod.default).forEach(([key, value]) => {
    50→      if (typeof value === "boolean") acc[key] = value;
    51→    });
    52→  }
    53→  return acc;
    54→}, {} as LocalApiOverrides);
    55→
    56→const STORAGE_SESSION_KEY = "localApi.override._serverSession";
    57→
    58→const syncFileOverridesToStorage = (): void => {
    59→  if (!IS_LOCAL_ENV) return;
    60→
    61→  try {
    62→    const storedSession = localStorage.getItem(STORAGE_SESSION_KEY);
    63→    if (storedSession === __SERVER_SESSION_ID__) return;
    64→
    65→    Object.entries(LOCAL_API_FILE_OVERRIDES).forEach(([key, value]) => {
    66→      localStorage.setItem(`localApi.override.${key}`, String(value));
    67→    });
    68→    localStorage.setItem(STORAGE_SESSION_KEY, __SERVER_SESSION_ID__);
    69→  } catch {
    70→    // localStorage unavailable
    71→  }
    72→};
    73→
    74→syncFileOverridesToStorage();
